33 template <class T>
34 Double_t Integrator::Integrate(T &f, const Double_t &a, const Double_t &b) {
35  Double_t m, h, fa, fb, i1, i2, is, erri1, erri2, r, y[13];
36  m = 0.5*(a + b);
37  h = 0.5*(b - a);
38  fa = y[0] = f(a);
39  fb = y[12] = f(b);
40  for (Int_t i = 1; i < 12; ++i) y[i] = f(m + x[i]*h);
41  i2 = (h/6.0)*(y[0] + y[12] + 5.0*(y[4] + y[8]));
42  i1 = (h/1470.0)*(77.0*(y[0] + y[12]) + 432.0*(y[2] + y[10]) +
43  625.0*(y[4] + y[8]) + 672.0*y[6]);
44  is = h*(0.0158271919734802*(y[0] + y[12]) + 0.0942738402188500*
45  (y[1] + y[11]) + 0.155071987336585*(y[2] + y[10]) +
46  0.188821573960182*(y[3] + y[9]) + 0.199773405226859*
47  (y[4] + y[8]) + 0.224926465333340*(y[5] + y[7]) +
48  0.242611071901408*y[6]);
49  erri1 = TMath::Abs(i1 - is);
50  erri2 = TMath::Abs(i2 - is);
51  r = (erri2 != 0.0) ? erri1/erri2 : 1.0;
52  toler = (r > 0.0 && r < 1.0) ? TOL/r : TOL;
53  if (is == 0.0) is = b - a;
54  is = TMath::Abs(is);
55  return int_helper(f, a, b, fa, fb, is);
56 }

58 template <class T>
59 Double_t Integrator::int_helper(T &f, const Double_t &a, const Double_t &b, const Double_t &fa,
60  const Double_t &fb, const Double_t &is) {
61  Double_t m, h, mll, ml, mr, mrr, fmll, fml, fm, fmrr, fmr, i1, i2;
62  m = 0.5*(a + b);
63  h = 0.5*(b - a);
64  mll = m - alpha*h;
65  ml = m - beta*h;
66  mr = m + beta*h;
67  mrr = m + alpha*h;
68  fmll = f(mll);
69  fml = f(ml);
70  fm = f(m);
71  fmr = f(mr);
72  fmrr = f(mrr);
73  i2 = h/6.0*(fa + fb + 5.0*(fml + fmr));
74  i1 = h/1470.0*(77.0*(fa + fb) + 432.0*(fmll + fmrr) + 625.0*(fml + fmr) + 672.0*fm);
75  if (TMath::Abs(i1 - i2) <= toler*is || mll <= a || b <= mrr) {
76  if ((mll <= a || b <= mrr) && terminate) {
77  out_of_tolerance=kTRUE;
78  terminate=kFALSE;
79  }
80  return i1;
81  }
82  else return int_helper(f, a, mll, fa, fmll, is) +
83  int_helper(f, mll, ml, fmll, fml, is) +
84  int_helper(f, ml, m, fml, fm, is) +
85  int_helper(f, m, mr, fm, fmr, is) +
86  int_helper(f, mr, mrr, fmr, fmrr, is) +
87  int_helper(f, mrr, b, fmrr, fb, is);
88 }
